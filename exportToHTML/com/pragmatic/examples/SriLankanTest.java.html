    private void typeAndSelect(String text, WebElement webElement) {
        Actions actions = new Actions(webDriver);
        actions.sendKeys(webElement, text)
                .pause(2000)
                .sendKeys(Keys.ARROW_DOWN)
                .pause(2000)
                .sendKeys(Keys.ENTER)
                .build()
                .perform();

    }
